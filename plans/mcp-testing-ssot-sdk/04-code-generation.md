# Phase 3: Code Generation

## Objective

Build the code generation engine in `src/lib/codegen.ts` that reads YAML tool definitions and shared schemas, then produces three committed files: `src/generated/types.ts` (TypeScript interfaces), `src/generated/validators.ts` (Zod schemas), and `src/generated/tool-list.ts` (tool name union + metadata map).

## Scope

### In Scope
- Code generation engine in `src/lib/codegen.ts`
- Full implementation of `scripts/generate.ts` with `--check` mode
- Generated output files committed to `src/generated/`
- Unit tests for the codegen engine
- npm script integration (`generate`, `generate:check`)

### Out of Scope
- Runtime usage of generated files (that happens in registry/handlers)
- Additional tool definitions beyond the 3 from Phase 2
- Test infrastructure (Phase 4)

## Step 1: Build the Code Generation Engine

Create `src/lib/codegen.ts`:

```typescript
import { jsonSchemaToZod } from 'json-schema-to-zod';
import type { ToolDefinition, JSONSchema7, SharedSchema } from './types.js';
import { resolveRefs } from './loader.js';

export interface CodegenInput {
  tools: ToolDefinition[];
  schemas: Map<string, SharedSchema>;
}

export interface CodegenOutput {
  types: string;
  validators: string;
  toolList: string;
}

function toTypeName(toolName: string): string {
  return toolName
    .split('_')
    .map((part) => part.charAt(0).toUpperCase() + part.slice(1))
    .join('');
}

function toSchemaName(typeName: string): string {
  return `${typeName}Schema`;
}

function jsonSchemaPropertyToTsType(prop: JSONSchema7, indent: number = 2): string {
  const pad = ' '.repeat(indent);

  if (prop.$ref) {
    const refName = prop.$ref.split('/').pop() ?? 'unknown';
    return refName;
  }

  if (prop.enum) {
    return prop.enum.map((v) => JSON.stringify(v)).join(' | ');
  }

  switch (prop.type) {
    case 'string':
      return 'string';
    case 'integer':
    case 'number':
      return 'number';
    case 'boolean':
      return 'boolean';
    case 'null':
      return 'null';
    case 'array': {
      if (prop.items) {
        const itemType = jsonSchemaPropertyToTsType(prop.items as JSONSchema7, indent);
        return `${itemType}[]`;
      }
      return 'unknown[]';
    }
    case 'object': {
      if (!prop.properties) {
        return 'Record<string, unknown>';
      }
      const required = new Set(prop.required ?? []);
      const lines: string[] = ['{'];
      for (const [key, value] of Object.entries(prop.properties)) {
        const optional = required.has(key) ? '' : '?';
        const tsType = jsonSchemaPropertyToTsType(value as JSONSchema7, indent + 2);
        lines.push(`${pad}  ${key}${optional}: ${tsType};`);
      }
      lines.push(`${pad}}`);
      return lines.join('\n');
    }
    default:
      return 'unknown';
  }
}

function generateInterfaceFromSchema(
  name: string,
  schema: JSONSchema7
): string {
  if (schema.type !== 'object' || !schema.properties) {
    return `export type ${name} = ${jsonSchemaPropertyToTsType(schema)};`;
  }

  const required = new Set(schema.required ?? []);
  const lines: string[] = [];
  lines.push(`export interface ${name} {`);

  for (const [key, value] of Object.entries(schema.properties)) {
    const prop = value as JSONSchema7;
    const optional = required.has(key) ? '' : '?';
    const tsType = jsonSchemaPropertyToTsType(prop, 2);
    if (prop.description) {
      lines.push(`  /** ${prop.description} */`);
    }
    lines.push(`  ${key}${optional}: ${tsType};`);
  }

  lines.push('}');
  return lines.join('\n');
}

export function generateTypes(input: CodegenInput): string {
  const lines: string[] = [];

  lines.push('// AUTO-GENERATED FILE — DO NOT EDIT');
  lines.push('// Generated by scripts/generate.ts from YAML tool definitions');
  lines.push('// Re-run with: npm run generate');
  lines.push('');

  const sharedTypeNames = new Set<string>();
  for (const [, schema] of input.schemas) {
    if (schema.definitions) {
      for (const [defName, defSchema] of Object.entries(schema.definitions)) {
        if (sharedTypeNames.has(defName)) continue;
        sharedTypeNames.add(defName);
        const resolved = resolveRefs(defSchema, input.schemas);
        lines.push(generateInterfaceFromSchema(defName, resolved));
        lines.push('');
      }
    }
  }

  lines.push('// --- Tool Input/Output Types ---');
  lines.push('');

  for (const tool of input.tools) {
    const typeName = toTypeName(tool.name);

    const resolvedInput = resolveRefs(tool.input, input.schemas);
    lines.push(generateInterfaceFromSchema(`${typeName}Input`, resolvedInput));
    lines.push('');

    const resolvedOutput = resolveRefs(tool.output, input.schemas);
    lines.push(generateInterfaceFromSchema(`${typeName}Output`, resolvedOutput));
    lines.push('');
  }

  return lines.join('\n');
}

export function generateValidators(input: CodegenInput): string {
  const lines: string[] = [];

  lines.push('// AUTO-GENERATED FILE — DO NOT EDIT');
  lines.push('// Generated by scripts/generate.ts from YAML tool definitions');
  lines.push('// Re-run with: npm run generate');
  lines.push('');
  lines.push("import { z } from 'zod';");
  lines.push('');

  for (const tool of input.tools) {
    const typeName = toTypeName(tool.name);

    const resolvedInput = resolveRefs(tool.input, input.schemas);
    const inputZodStr = jsonSchemaToZod(resolvedInput as Record<string, unknown>, {
      module: 'none',
      name: toSchemaName(`${typeName}Input`),
      type: true,
    });
    lines.push(inputZodStr);
    lines.push('');

    const resolvedOutput = resolveRefs(tool.output, input.schemas);
    const outputZodStr = jsonSchemaToZod(resolvedOutput as Record<string, unknown>, {
      module: 'none',
      name: toSchemaName(`${typeName}Output`),
      type: true,
    });
    lines.push(outputZodStr);
    lines.push('');
  }

  return lines.join('\n');
}

export function generateToolList(input: CodegenInput): string {
  const lines: string[] = [];

  lines.push('// AUTO-GENERATED FILE — DO NOT EDIT');
  lines.push('// Generated by scripts/generate.ts from YAML tool definitions');
  lines.push('// Re-run with: npm run generate');
  lines.push('');

  const toolNames = input.tools.map((t) => `'${t.name}'`);
  lines.push(`export type ToolName = ${toolNames.join(' | ')};`);
  lines.push('');

  lines.push(`export const TOOL_NAMES: readonly ToolName[] = [${toolNames.join(', ')}] as const;`);
  lines.push('');

  lines.push('export interface ToolMetadata {');
  lines.push('  name: ToolName;');
  lines.push('  description: string;');
  lines.push('  permission: string;');
  lines.push('  handler: string;');
  lines.push('  cacheTtl?: number;');
  lines.push('}');
  lines.push('');

  lines.push('export const TOOL_METADATA: Record<ToolName, ToolMetadata> = {');
  for (const tool of input.tools) {
    lines.push(`  '${tool.name}': {`);
    lines.push(`    name: '${tool.name}',`);
    lines.push(`    description: ${JSON.stringify(tool.description)},`);
    lines.push(`    permission: '${tool.permission}',`);
    lines.push(`    handler: '${tool.handler}',`);
    if (tool.cache) {
      lines.push(`    cacheTtl: ${tool.cache.ttl},`);
    }
    lines.push('  },');
  }
  lines.push('};');
  lines.push('');

  lines.push(`export const TOOL_COUNT = ${input.tools.length} as const;`);
  lines.push('');

  return lines.join('\n');
}

export function generateAll(input: CodegenInput): CodegenOutput {
  return {
    types: generateTypes(input),
    validators: generateValidators(input),
    toolList: generateToolList(input),
  };
}
```

**IMPORTANT**: The `json-schema-to-zod` library's API may vary. The `module: 'none'` option (if supported) generates standalone code without import/export. If this option does not exist, the codegen must strip the import line and export keyword from the output. Test this during implementation.

## Step 2: Implement the Generate Script

Replace `scripts/generate.ts` (stub created in Phase 0):

```typescript
#!/usr/bin/env npx tsx

import { readFileSync, writeFileSync, existsSync, mkdirSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import { loadToolsFromDirectory } from '../src/lib/loader.js';
import { generateAll } from '../src/lib/codegen.js';

const __dirname = dirname(fileURLToPath(import.meta.url));
const PROJECT_ROOT = join(__dirname, '..');

const DEFINITIONS_DIR = join(PROJECT_ROOT, 'src/tools/definitions');
const SCHEMAS_DIR = join(PROJECT_ROOT, 'src/tools/schemas');
const GENERATED_DIR = join(PROJECT_ROOT, 'src/generated');

const isCheckMode = process.argv.includes('--check');

function main(): void {
  console.log('Loading tool definitions...');

  const loadResult = loadToolsFromDirectory(DEFINITIONS_DIR, SCHEMAS_DIR);

  if (loadResult.errors.length > 0) {
    console.error('Errors loading tool definitions:');
    for (const error of loadResult.errors) {
      console.error(`  ${error.file}: ${error.message}`);
    }
    process.exit(1);
  }

  console.log(`Loaded ${loadResult.tools.length} tool definitions`);
  console.log(`Loaded ${loadResult.schemas.size} shared schemas`);

  const tools = loadResult.tools.map((t) => t.definition);
  const output = generateAll({ tools, schemas: loadResult.schemas });

  const files: Record<string, string> = {
    'types.ts': output.types,
    'validators.ts': output.validators,
    'tool-list.ts': output.toolList,
  };

  if (!existsSync(GENERATED_DIR)) {
    mkdirSync(GENERATED_DIR, { recursive: true });
  }

  if (isCheckMode) {
    let hasChanges = false;

    for (const [filename, content] of Object.entries(files)) {
      const filePath = join(GENERATED_DIR, filename);

      if (!existsSync(filePath)) {
        console.error(`MISSING: ${filePath}`);
        hasChanges = true;
        continue;
      }

      const existing = readFileSync(filePath, 'utf-8');
      if (existing !== content) {
        console.error(`OUT OF DATE: ${filePath}`);
        hasChanges = true;
      } else {
        console.log(`OK: ${filePath}`);
      }
    }

    if (hasChanges) {
      console.error('\nGenerated files are out of date. Run: npm run generate');
      process.exit(1);
    }

    console.log('\nAll generated files are up to date.');
    return;
  }

  for (const [filename, content] of Object.entries(files)) {
    const filePath = join(GENERATED_DIR, filename);
    writeFileSync(filePath, content, 'utf-8');
    console.log(`Generated: ${filePath}`);
  }

  const gitkeep = join(GENERATED_DIR, '.gitkeep');
  if (existsSync(gitkeep)) {
    const { unlinkSync } = await import('node:fs');
    unlinkSync(gitkeep);
    console.log(`Removed: ${gitkeep}`);
  }

  console.log(`\nCode generation complete. ${Object.keys(files).length} files written.`);
}

main();
```

**CRITICAL**: The `--check` mode is essential for CI. It exits with code 1 if generated files are out of date, ensuring developers run `npm run generate` after modifying YAML definitions.

**IMPORTANT**: The script uses `await import()` for the unlinkSync call which requires top-level await. Since this is an ESM script running under tsx, top-level await is supported. However, if there are issues, wrap `main()` as an async function.

import { describe, it, expect } from 'vitest';
import {
  generateTypes,
  generateValidators,
  generateToolList,
  generateAll,
} from '@lib/codegen.js';
import type { CodegenInput } from '@lib/codegen.js';
import type { ToolDefinition, SharedSchema } from '@lib/types.js';

const MOCK_TOOL: ToolDefinition = {
  name: 'test_tool',
  description: 'A test tool for unit testing code generation',
  permission: 'read_pr',
  cache: { ttl: 300, key: '${workspace}:test' },
  input: {
    type: 'object',
    required: ['workspace'],
    properties: {
      workspace: { type: 'string', description: 'Workspace slug' },
      limit: { type: 'integer', minimum: 1 },
    },
  },
  output: {
    type: 'object',
    properties: {
      items: { type: 'array', items: { type: 'string' } },
      count: { type: 'integer' },
    },
    required: ['items', 'count'],
  },
  handler: 'src/handlers/test.ts#handleTest',
};

const MOCK_SHARED: SharedSchema = {
  definitions: {
    SimpleUser: {
      type: 'object',
      properties: {
        id: { type: 'string' },
        name: { type: 'string' },
      },
      required: ['id', 'name'],
    },
  },
};

function makeInput(tools?: ToolDefinition[], schemas?: Map<string, SharedSchema>): CodegenInput {
  return {
    tools: tools ?? [MOCK_TOOL],
    schemas: schemas ?? new Map([['common', MOCK_SHARED]]),
  };
}

describe('generateTypes', () => {
  it('generates auto-generated header', () => {
    const output = generateTypes(makeInput());
    expect(output).toContain('AUTO-GENERATED FILE');
    expect(output).toContain('DO NOT EDIT');
  });

  it('generates shared schema interfaces', () => {

    expect(output).toContain('export interface SimpleUser');

  it('generates tool input interface', () => {

    expect(output).toContain('export interface TestToolInput');
    expect(output).toContain('workspace: string');
  });

  it('generates tool output interface', () => {
    const output = generateTypes(makeInput());
    expect(output).toContain('export interface TestToolOutput');
    expect(output).toContain('items: string[]');

    expect(output.toolList.length).toBeGreaterThan(0);
  });
